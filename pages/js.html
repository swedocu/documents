    /*Personel id si  3 ve 3ten büyük olanların ad soyad ve maaş bilgilerini görüntüleyen sorgu*/
            
    select ADI, SOYADI, MAAŞ  from personelT where PER_id >= 3
            
    /*Görevi mühendis şehri ankara olanların maaşının yüzde yirmisi sogrulama*/
            
    select ADI, SOYADI, MAAŞ*0.2 as YÜZDEYİRMİ, GÖREVI from personelT where GÖREVI = 'Mühendis' AND  ŞEHIR = 'Ankara'
            
    /*Soyadı Demir olan Ya da maaşı 5000 den fazla olup şehri ankara olan....*/
        
    select * from personelT where SOYADI = 'Demir' OR (MAAŞ > 5000 and ŞEHIR = 'Ankara')
